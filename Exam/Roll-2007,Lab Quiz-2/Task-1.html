function countVowels(sentence) { // Define vowels (both lowercase and uppercase) const vowels = ['a', 'e', 'i', 'o', 'u']; let count = 0; // Convert sentence to lowercase to count vowels case-insensitively const lowerSentence = sentence.toLowerCase(); // Loop through each character in the sentence for (let char of lowerSentence) { if (vowels.includes(char)) { count++; } } return count; } // Example usage: const input = "Hello! How are you?"; const vowelCount = countVowels(input); console.log(`There are ${vowelCount} vowels in the sentence.`);
